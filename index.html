function averageOfNumbers(arr) { //числові елементи масиву const numbers = arr.filter(item => typeof item === 'number'); //числових елементів немає, повертаємо 0, щоб уникнути ділення на 0 if (numbers.length === 0) return 0; //сумa const sum = numbers.reduce((acc, num) => acc + num, 0); //середнє арифметичне return sum / numbers.length; } // Приклад const array = [1, 'hello', 3, true, 5, null, 10]; const result = averageOfNumbers(array); console.log(result); // Виведе 4.75
