function removeChars(inputString, charsToRemove) { // Створюємо множину з символів const charsSet = new Set(charsToRemove); // Фільтруємо символи рядка let result = ''; for (let char of inputString) { if (!charsSet.has(char)) { result += char; } } return result; } // використання const output = removeChars(" hello world", ['l', 'd']); console.log(output); // Виводіть "heo wor"g
